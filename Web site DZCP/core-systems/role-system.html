نظام الأدوار في DZCP

مقدمة

يسمح لك نظام الأدوار بإنشاء أدوار مخصصة مع سلوكيات وخواص فريدة.

إنشاء دور مخصص


                        using DZCP.API;
using DZCP.Roles;

public class ScientistGuardRole : CustomRole
{
    public override string RoleName => "Scientist Guard";
    public override string Description => "حارس مختص بحماية العلماء";
    public override TeamType Team => TeamType.MTF;

    public override void OnSpawn(Player player)
    {
        // منح المعدات عند الظهور
        player.Inventory.Add(ItemType.GunCOM15);
        player.Inventory.Add(ItemType.Medkit);

        // إرسال رسالة ترحيبية
        player.SendMessage("أنت الآن حارس عالم!");
    }
}

تسجيل الدور

لجعل الدور متاحًا في اللعبة:

public class MyPlugin : IPlugin
{
    public void OnEnabled()
    {
        RoleManager.RegisterRole(new ScientistGuardRole());
    }
}

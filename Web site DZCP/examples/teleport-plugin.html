using DZCP.API;

    {
        CommandManager.RegisterCommand("tp", "ينقل اللاعب إلى هدف", OnTeleportCommand);
    }

    {
        // سيتم تنفيذ الكود هنا

الخطوة 3: تنفيذ الأمر

private void OnTeleportCommand(Player player, string[] args)
{
    if (args.Length < 1)
    {
        player.SendMessage("استخدم: .tp [اسم اللاعب]");
        return;
    }

    Player target = Player.Get(args[0]);
    if (target != null)
    {
        player.Teleport(target.Position);
        player.SendMessage($"تم نقلك إلى {target.Nickname}");
    }
    else
    {
        player.SendMessage("اللاعب غير موجود!");
    }
}
